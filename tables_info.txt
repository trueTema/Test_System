users (
    id int primary key
    name varchar(128) not null
    status varchar(16) not null
    study_group varchar(32) not null
)

parcels (
    problem_ID int primary key
    student_ID int check(student_ID > 0) not null unique
    points float not null
    answer text not null
    sending_time big_int check(sending_time > 0)
)

problems (
    problem_ID int primary key
    teacher_ID int not null check(teacher_ID > 0) unique
    problem_situation text
    is_visible int check(is_visible in (0, 1)) default 1
    deadline int not null check(deadline >0)
    users_group varchar(32)
)
